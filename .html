// JavaScript version of 'Guess the Number' game function guessTheNumber() { const randomNumber = Math.floor(Math.random() * 100) + 1; const maxAttempts = 5; console.log("Welcome to 'Guess the Number' game!"); console.log("I'm thinking of a number between 1 and 100. You have 5 chances to guess it!"); for (let attempt = 1; attempt <= maxAttempts; attempt++) { let guess = prompt(`Attempt ${attempt}/${maxAttempts}: Enter your guess:`); if (guess === null) { console.log("Game cancelled."); return; } guess = parseInt(guess); if (isNaN(guess)) { console.log("Please enter a valid number."); attempt--; continue; } if (guess === randomNumber) { console.log(`Congratulations! You guessed the correct number: ${randomNumber}`); break; } else if (guess < randomNumber) { console.log("Your guess is too low."); } else { console.log("Your guess is too high."); } if (attempt < maxAttempts) { console.log(`You have ${maxAttempts - attempt} guess(es) left. Try again!`); } else { console.log(`Sorry, you've used all your chances. The correct number was: ${randomNumber}`); } } } guessTheNumber();
